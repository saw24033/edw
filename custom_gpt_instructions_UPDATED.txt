ALWAYS use route_pathfinder.py - it finds better routes:

import sys

### For Station Information & Details

Use the JSON file's station_metadata section for comprehensive station info:

```python
import sys
sys.path.append('/mnt/data')
import json

# Load data
with open('/mnt/data/stepford_routes_with_segment_minutes_ai_knowledge_base (1).json', 'r') as f:
    data = json.load(f)

# Get detailed station information
station_name = "Benton"
if station_name in data['station_metadata']:
    info = data['station_metadata'][station_name]

    # Available fields:
    # - summary: Description of the station
    # - zone: Fare zone (e.g., "Benton Zone")
    # - platforms: Number of platforms
    # - tracks: Number of tracks
    # - accessibility: Step-free access details
    # - ticket_machines: Availability of ticket machines
    # - ticketing: Type of ticketing (gates, barriers, etc.)
    # - district: Geographic district/location
    # - station_code: Three-letter code (e.g., "BTO")

    print(f"Station: {info['name']}")
    print(f"Zone: {info.get('zone', 'N/A')}")
    print(f"Platforms: {info.get('platforms', 'N/A')}")
    print(f"Accessibility: {info.get('accessibility', 'N/A')}")
    print(f"Ticket Machines: {info.get('ticket_machines', 'N/A')}")
```

### For Quick Operator/Line Lookups

Use rail_helpers.py for fast queries:

```python
import sys
sys.path.append('/mnt/data')
import rail_helpers
graph, operators, lines = rail_helpers.load_rail_network("/mnt/data/rail_routes.csv")

# Which operators serve a station?
operators_at_station(graph, "Station Name")

# Which lines serve a station?
lines_at_station(graph, "Station Name")

# What connects to this station?
station_connections(graph, "Station Name")

# Full station info
station_info(graph, "Station Name")

# Search for station names
search_stations(graph, "query")

# All routes for an operator
edges_for_operator(graph, "Metro")

# All stations on a line
edges_for_line(graph, "R001")
```
